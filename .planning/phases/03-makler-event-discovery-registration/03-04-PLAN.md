    **Separating active vs past/full:**
    - Active section: Events where state==Public AND not full (regardless of registration status)
    - Also active: Events where user is registered AND event is not Finished (user's upcoming events)

    **Code-behind (@code block):**

    ```csharp
    private List<Event> allEvents = new();
    private string searchTerm = "";
    private string dateFilter = "quarter"; // default: Nächste 3 Monate
    private bool isLoading = true;
    private string userEmail = "";
    private System.Threading.Timer? debounceTimer;

    protected override async Task OnInitializedAsync()
    {
        var authState = await AuthStateProvider.GetAuthenticationStateAsync();
        userEmail = authState.User.FindFirst("preferred_username")?.Value
                    ?? authState.User.Identity?.Name ?? "";
        await LoadEventsAsync();
        isLoading = false;
    }

    private void OnSearchChanged()
    {
        debounceTimer?.Dispose();
        debounceTimer = new System.Threading.Timer(_ =>
        {
            InvokeAsync(async () =>
            {
                await LoadEventsAsync();
                StateHasChanged();
            });
        }, null, 300, Timeout.Infinite);
    }

    private async Task SetDateFilter(string filter)
    {
        dateFilter = filter;
        await LoadEventsAsync();
    }

    private async Task LoadEventsAsync()
    {
        var now = DateTime.UtcNow;
        DateTime? startDateTo = dateFilter switch
        {
            "month" => now.AddMonths(1),
            "quarter" => now.AddMonths(3),
            "year" => now.AddYears(1),
            _ => now.AddMonths(3)
        };

        allEvents = await EventService.GetPublicEventsAsync(
            searchTerm: string.IsNullOrWhiteSpace(searchTerm) ? null : searchTerm,
            startDateFrom: null, // include all from past (for the collapsed section)
            startDateTo: startDateTo,
            userEmail: userEmail);
    }

    private bool IsUserRegistered(Event evt)
        => evt.Registrations.Any(r => r.Email == userEmail && !r.IsCancelled);

    private bool IsEventActive(Event evt)
    {
        var state = evt.GetCurrentState();
        var isRegistered = IsUserRegistered(evt);
        var isFull = evt.GetCurrentRegistrationCount() >= evt.MaxCapacity;

        // Active: upcoming public events, or user's registered upcoming events
        if (state == EventState.Public && !isFull) return true;
        if (isRegistered && state != EventState.Finished) return true;
        return false;
    }

    private List<Event> GetActiveEvents()
        => allEvents.Where(IsEventActive).OrderBy(e => e.StartDateUtc).ToList();

    private List<Event> GetPastOrFullEvents()
        => allEvents.Where(e => !IsEventActive(e)).OrderByDescending(e => e.StartDateUtc).ToList();

    public void Dispose()
    {
        debounceTimer?.Dispose();
    }
    ```
