```csharp
private void OnEventTypeChanged()
{

        Model.MaxCapacity = 0;
        Model.MaxCompanions = 0;
        Model.AgendaItems.Clear();
        agendaItemDatesCet.Clear();
        Model.EventOptions.Clear();
    }
}
```
