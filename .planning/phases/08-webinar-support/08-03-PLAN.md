7. Fix IsEventActive to handle webinars (webinars are never "full" or "past deadline"):
```csharp
private bool IsEventActive(Event evt)
{
    // Webinars: always active if published (no capacity/deadline constraints)
    if (evt.EventType == EventType.Webinar)
    {
        return evt.GetCurrentState() != EventState.Finished;
    }

    var state = evt.GetCurrentState();
    var isRegistered = IsUserRegistered(evt);
    var isFull = evt.GetCurrentRegistrationCount() >= evt.MaxCapacity;

    if (state == EventState.Public && !isFull) return true;
    if (isRegistered && state != EventState.Finished) return true;
    return false;
}
```
