**Step 3: Implement @code block.**

```csharp
@code {
    [Parameter] public int EventId { get; set; }

    private Event? evt;
    private List<Registration> allParticipants = new();
    private List<Registration> filteredParticipants = new();
    private string companyFilter = "";
    private bool isExporting = false;

    protected override async Task OnInitializedAsync()
    {
        evt = await EventService.GetEventByIdAsync(EventId);
        if (evt == null)
        {
            NavigationManager.NavigateTo("/admin/events");
            return;
        }
        allParticipants = await ParticipantQueryService.GetParticipantsAsync(EventId);
        filteredParticipants = allParticipants;
    }

    private void ApplyFilter()
    {
        if (string.IsNullOrWhiteSpace(companyFilter))
        {
            filteredParticipants = allParticipants;
        }
        else
        {
            filteredParticipants = allParticipants
                .Where(p => GetCompanyName(p).Contains(companyFilter, StringComparison.OrdinalIgnoreCase))
                .ToList();
        }
    }

    private void ClearFilter()
    {
        companyFilter = "";
        filteredParticipants = allParticipants;
    }

    private string GetCompanyName(Registration r)
    {
        return r.EventCompany?.CompanyName ?? r.Company ?? "N/A";
    }

    // Export methods using JS interop to trigger file download
    private async Task ExportParticipantList() => await DownloadExcel(
        () => ParticipantExportService.ExportParticipantListAsync(EventId),
        $"Teilnehmerliste_{evt!.Title}_{DateTime.Now:yyyyMMdd}.xlsx");

    private async Task ExportContactData() => await DownloadExcel(
        () => ParticipantExportService.ExportContactDataAsync(EventId),
        $"Kontaktdaten_{evt!.Title}_{DateTime.Now:yyyyMMdd}.xlsx");

    private async Task ExportNonParticipants() => await DownloadExcel(
        () => ParticipantExportService.ExportNonParticipantsAsync(EventId),
        $"NichtTeilnehmer_{evt!.Title}_{DateTime.Now:yyyyMMdd}.xlsx");

    private async Task ExportCompanyList() => await DownloadExcel(
        () => ParticipantExportService.ExportCompanyListAsync(EventId),
        $"Firmenliste_{evt!.Title}_{DateTime.Now:yyyyMMdd}.xlsx");

    private async Task DownloadExcel(Func<Task<byte[]>> exportFunc, string fileName)
    {
        isExporting = true;
        try
        {
            var bytes = await exportFunc();
            // Use JS interop to trigger browser download
            // Use Blazor's built-in approach: convert to base64, create data URL, trigger click
            var base64 = Convert.ToBase64String(bytes);
            await JSRuntime.InvokeVoidAsync("downloadFile", fileName,
                "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet", base64);
        }
        finally
        {
            isExporting = false;
        }
    }
}
```

**Step 4: Add JavaScript helper for file download.**
Check if a `downloadFile` JS function already exists in the project. If not, add it to the main layout or a shared JS file. The simplest approach for Blazor Server is to add a small script in `App.razor` or `_Host.cshtml`:

```javascript
window.downloadFile = (fileName, contentType, base64) => {
    const link = document.createElement('a');
    link.download = fileName;
    link.href = `data:${contentType};base64,${base64}`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
};
```
