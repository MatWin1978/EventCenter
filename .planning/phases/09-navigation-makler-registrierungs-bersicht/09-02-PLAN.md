    @code block:
    ```csharp
    @code {
        private List<Registration> registrations = new();
        private bool isLoading = true;

        protected override async Task OnInitializedAsync()
        {
            var authState = await AuthStateProvider.GetAuthenticationStateAsync();
            var userEmail = authState.User.FindFirst("preferred_username")?.Value
                            ?? authState.User.Identity?.Name ?? string.Empty;

            registrations = await RegistrationService.GetBrokerRegistrationsAsync(userEmail);
            isLoading = false;
        }

        private decimal GetTotalCost(Registration reg)
        {
            var agendaCost = reg.RegistrationAgendaItems.Sum(rai => rai.AgendaItem.CostForMakler);
            var optionsCost = reg.SelectedOptions.Sum(o => o.Price);
            return agendaCost + optionsCost;
        }
    }
    ```
